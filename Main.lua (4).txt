local function HQQZ_fake_script() -- BG.LocalScript 
	local script = Instance.new('LocalScript', BG)

	local frame = script.Parent
	local UserInputService = game:GetService("UserInputService")
	
	local dragging
	local dragInput
	local dragStart
	local startPos
	
	local function update(input)
		local delta = input.Position - dragStart
		frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
	end
	
	frame.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragStart = input.Position
			startPos = frame.Position
	
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false
				end
			end)
		end
	end)
	
	frame.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput = input
		end
	end)
	
	UserInputService.InputChanged:Connect(function(input)
		if input == dragInput and dragging then
			update(input)
		end
	end)
end
coroutine.wrap(HQQZ_fake_script)()
local function XHDMF_fake_script() -- BG.LocalScript 
	local script = Instance.new('LocalScript', BG)

	local Players = game:GetService("Players")
	local RunService = game:GetService("RunService")
	local player = Players.LocalPlayer
	local frontFlipButton = script.Parent:FindFirstChild("FrontFlipButton")
	local backFlipButton = script.Parent:FindFirstChild("BackFlipButton")
	local flipEnabled = false
	local inAir = false
	local flipped = false
	local humanoid
	local hrp
	local stateConnection
	local renderConnection
	local currentFlipType = "none"
	local onCooldown = false
	local COOLDOWN_TIME = 1
	local activeBAV = nil
	
	local function setupCharacter(character)
		humanoid = character:WaitForChild("Humanoid")
		hrp = character:WaitForChild("HumanoidRootPart")
	
		local function onStateChanged(_, state)
			if not flipEnabled then return end
			if state == Enum.HumanoidStateType.Freefall then
				inAir = true
				flipped = false
			elseif state == Enum.HumanoidStateType.Landed then
				inAir = false
				if activeBAV and activeBAV.Parent then
					activeBAV:Destroy()
					activeBAV = nil
				end
				local currentCFrame = hrp.CFrame
				local uprightCFrame = CFrame.new(currentCFrame.Position) * CFrame.Angles(0, math.atan2(-currentCFrame.LookVector.X, -currentCFrame.LookVector.Z), 0)
				hrp.CFrame = uprightCFrame
				hrp.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
				humanoid.AutoRotate = true
				flipEnabled = false
				currentFlipType = "none"
			end
		end
	
		if stateConnection then
			stateConnection:Disconnect()
		end
		if renderConnection then
			renderConnection:Disconnect()
		end
	
		stateConnection = humanoid.StateChanged:Connect(onStateChanged)
	
		renderConnection = RunService.RenderStepped:Connect(function()
			if not character.Parent then
				stateConnection:Disconnect()
				renderConnection:Disconnect()
				return
			end
	
			if flipEnabled and inAir and not flipped and currentFlipType ~= "none" then
				local vy = hrp.AssemblyLinearVelocity.Y
				if vy <= 0 then
					flipped = true
					local chosenFlipType = currentFlipType
					task.spawn(function()
						if activeBAV and activeBAV.Parent then
							activeBAV:Destroy()
						end
						humanoid.AutoRotate = false
						local jumpHeight = math.abs(vy) * 0.25
						local flipDuration = math.clamp(jumpHeight, 0.2, 0.5)
						local angVel = 2 * math.pi / flipDuration
						local bav = Instance.new("BodyAngularVelocity")
						bav.MaxTorque = Vector3.new(math.huge, math.huge, math.huge)
						bav.P = 5000
						if chosenFlipType == "front" then
							bav.AngularVelocity = -hrp.CFrame.RightVector * angVel
						elseif chosenFlipType == "back" then
							bav.AngularVelocity = hrp.CFrame.RightVector * angVel
						end
						bav.Parent = hrp
						activeBAV = bav
						task.wait(flipDuration)
						if bav and bav.Parent then
							bav:Destroy()
						end
						activeBAV = nil
						hrp.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
						local currentCFrame = hrp.CFrame
						local uprightCFrame = CFrame.new(currentCFrame.Position) * CFrame.Angles(0, math.atan2(-currentCFrame.LookVector.X, -currentCFrame.LookVector.Z), 0)
						hrp.CFrame = uprightCFrame
						humanoid.AutoRotate = true
					end)
				end
			end
		end)
	end
	
	if player.Character then
		setupCharacter(player.Character)
	end
	player.CharacterAdded:Connect(setupCharacter)
	
	local function activateFlip(flipType)
		if onCooldown then return end
		if humanoid and humanoid.Parent then
			if activeBAV and activeBAV.Parent then
				activeBAV:Destroy()
				activeBAV = nil
			end
			hrp.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
			currentFlipType = flipType
			flipEnabled = true
			inAir = false
			flipped = false
			humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
			onCooldown = true
			task.delay(COOLDOWN_TIME, function()
				onCooldown = false
			end)
		end
	end
	
	if frontFlipButton then
		frontFlipButton.MouseButton1Click:Connect(function()
			activateFlip("front")
		end)
	end
	
	if backFlipButton then
		backFlipButton.MouseButton1Click:Connect(function()
			activateFlip("back")
		end)
	end
end

local function EMMFNCR_fake_script() -- FrontFlipButton.FrontFlip 
	local script = Instance.new('LocalScript', FrontFlipButton)

	-- ================================================================================
	-- Section used for FrontFlip
	-- ========================================
	
	
	local Players = game:GetService("Players")
	local RunService = game:GetService("RunService")
	local player = Players.LocalPlayer
	local button = script.Parent
	
	local flipEnabled = false
	local inAir = false
	local flipped = false
	local humanoid
	local hrp
	local stateConnection
	local renderConnection
	
	local function setupCharacter(character)
		humanoid = character:WaitForChild("Humanoid")
		hrp = character:WaitForChild("HumanoidRootPart")
	
		local function onStateChanged(_, state)
			if not flipEnabled then return end
	
			if state == Enum.HumanoidStateType.Freefall then
				inAir = true
				flipped = false
			elseif state == Enum.HumanoidStateType.Landed then
				inAir = false
				humanoid.AutoRotate = true
				flipEnabled = false
			end
		end
	
		if stateConnection then
			stateConnection:Disconnect()
		end
		if renderConnection then
			renderConnection:Disconnect()
		end
	
		stateConnection = humanoid.StateChanged:Connect(onStateChanged)
	
		renderConnection = RunService.RenderStepped:Connect(function()
			if not character.Parent then
				-- Character was destroyed, cleanup connections
				stateConnection:Disconnect()
				renderConnection:Disconnect()
				return
			end
	
			if flipEnabled and inAir and not flipped then
				local vy = hrp.AssemblyLinearVelocity.Y
				if vy <= 0 then -- apex
					flipped = true
					task.spawn(function()
						humanoid.AutoRotate = false
						-- Flip duration based on jump height
						local jumpHeight = math.abs(vy) * 0.25
						local flipDuration = math.clamp(jumpHeight, 0.2, 0.5)
						local angVel = 2 * math.pi / flipDuration
						-- Front flip (negative RightVector for forward rotation)
						local bav = Instance.new("BodyAngularVelocity")
						bav.MaxTorque = Vector3.new(math.huge, math.huge, math.huge)
						bav.AngularVelocity = -hrp.CFrame.RightVector * angVel
						bav.P = 5000
						bav.Parent = hrp
						task.wait(flipDuration)
						-- Cleanup
						bav:Destroy()
	
						-- Align player upright for landing
						local currentCFrame = hrp.CFrame
						local uprightCFrame = CFrame.new(currentCFrame.Position) * CFrame.Angles(0, math.atan2(-currentCFrame.LookVector.X, -currentCFrame.LookVector.Z), 0)
						hrp.CFrame = uprightCFrame
	
						humanoid.AutoRotate = true
					end)
				end
			end
		end)
	end
	
	if player.Character then
		setupCharacter(player.Character)
	end
	
	player.CharacterAdded:Connect(setupCharacter)
	
	button.MouseButton1Click:Connect(function()
		if humanoid and humanoid.Parent then
			flipEnabled = true
			inAir = false
			flipped = false
			humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
		end
	end)
	
	-- ================================================================================
end
